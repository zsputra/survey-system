public class ResponseRowMapper implements RowMapper<Response> {

        response.setSurveyId(rs.getInt("response_survey_id"));
        response.setPageId(rs.getInt("response_page_id"));
        response.setQuestionId(rs.getInt("question_id"));
        response.setQuestionType(rs.getString("questionType"));
        response.setAnswer(rs.getString("answer"));
        response.setDateAnswer(rs.getTimestamp("dateanswer"));
